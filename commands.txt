# Install only the absolute bare essentials to your target mount point (/mnt)
XBPS_ARCH=$ARCH xbps-install -R "$REPO" -S -r /mnt base-files coreutils runit xbps kmod linux

xbps-query -Rf linux6.6
xbps-query -Rx base-system

# custom kernel without sound modules.


## installation commands
cfdisk /dev/sda

mkfs.vfat /dev/sda1
mkfs.ext4 /dev/sda2
mount /dev/sda2 /mnt/
mkdir -p /mnt/boot/efi/
mount /dev/sda1 /mnt/boot/efi/

xgenfstab -U /mnt > /mnt/etc/fstab

# Install packages from packages.list into /mnt
mkdir -p /mnt/var/db/xbps/keys
cp /var/db/xbps/keys/* /mnt/var/db/xbps/keys/
REPO=https://repo-default.voidlinux.org/current
ARCH=x86_64
XBPS_ARCH=$ARCH xbps-install -S -R "$REPO" -r /mnt
grep -v '^\s*#' packages.list | grep -v '^\s*$' | xargs XBPS_ARCH=$ARCH xbps-install -r /mnt

xchroot /mnt /bin/bash
xbps-reconfigure -f glibc-locales
passwd

mount -t efivarfs none /sys/firmware/efi/efivars
grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id="Void"

xbps-reconfigure -fa

exit
umount -R /mnt
shutdown -r now
