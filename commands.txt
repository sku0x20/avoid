# Install only the absolute bare essentials to your target mount point (/mnt)
XBPS_ARCH=$ARCH xbps-install -R "$REPO" -S -r /mnt base-files coreutils runit xbps kmod linux

xbps-query -f linux6.6
xbps-query -Rf linux6.6
xbps-query -x base-system

# Print package details; detect meta-package (no files) and show deps if so
pkg=base-system
xbps-query -RS "$pkg"
files=$(xbps-query -Rf "$pkg")
if [ -z "$files" ]; then
    echo "--- meta-package: dependencies ---"
    xbps-query -Rx "$pkg"
fi
